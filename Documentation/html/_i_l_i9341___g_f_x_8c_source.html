211void ILI9341_DrawChar(char ch, const uint8_t font[], uint16_t X, uint16_t Y, uint16_t color, uint16_t bgcolor)
212{
213 if ((ch < 31) || (ch > 127)) return;
214
215 uint8_t fOffset, fWidth, fHeight, fBPL;
216 uint8_t *tempChar;
217
218 fOffset = font[0];
219 fWidth = font[1];
220 fHeight = font[2];
221 fBPL = font[3];
222
223 tempChar = (uint8_t*)&font[((ch - 0x20) * fOffset) + 4]; /* Current Character = Meta + (Character Index * Offset) */
224
225 /* Clear background first */
226 ILI9341_DrawRectangle(X, Y, fWidth, fHeight, bgcolor);
227
228 for (int j=0; j < fHeight; j++)
229 {
230 for (int i=0; i < fWidth; i++)
231 {
232 uint8_t z = tempChar[fBPL * i + ((j & 0xF8) >> 3) + 1]; /* (j & 0xF8) >> 3, increase one by 8-bits */
233 uint8_t b = 1 << (j & 0x07);
234 if (( z & b ) != 0x00)
235 {
236 ILI9341_DrawPixel(X+i, Y+j, color);
237 }
238 }
239 }

251void ILI9341_DrawText(const char* str, const uint8_t font[], uint16_t X, uint16_t Y, uint16_t color, uint16_t bgcolor)
252{
253 uint8_t charWidth; /* Width of character */
254 uint8_t fOffset = font[0]; /* Offset of character */
255 uint8_t fWidth = font[1]; /* Width of font */
256
257 while (*str)
258 {
259 ILI9341_DrawChar(*str, font, X, Y, color, bgcolor);
260
261 /* Check character width and calculate proper position */
262 uint8_t *tempChar = (uint8_t*)&font[((*str - 0x20) * fOffset) + 4];
263 charWidth = tempChar[0];
264
265 if(charWidth + 2 < fWidth)
266 {
267 /* If character width is smaller than font width */
268 X += (charWidth + 2);
269 }
270 else

272 X += fWidth;
273 }
